  kmeans_model <- kmeans(DADOS_POP, centers = i)  # EXECUTA O ALGORITMO DE K-MEANS
  sse[i] <- kmeans_model$tot.withinss  #ARMAZENAR A SOMA DO ERRO QUADRATICO (SSE) PARA O VALOR DE K ATUAL
